# 1. Generate seed via Setoworks API curl -X POST https://api.setoworks.com/v1/lookalike/seed \ -H "Authorization: Bearer sk_live_xxx" \ -d '{"category":"tech_gadgets","price_min":50,"price_max":500,"platform":"meta","format":"meta_audience"}' # Response: { "audience_url": "https://...meta.csv", "size": 12309 } # 2. Push to Meta Custom Audience (auto) curl -X POST https://graph.facebook.com/v19.0/act_/customaudiences \ -d "name=Setoworks_Lookalike_Tech_2026" \ -d "subtype=CUSTOM" \ -d "customer_file_source=USER_PROVIDED_ONLY" \ -d "access_token=" # 3. Create lookalike (Meta auto-expands to ~10M lookalikes) curl -X POST https://graph.facebook.com/v19.0/act_/customaudiences \ -d "subtype=LOOKALIKE" \ -d 'lookalike_spec={"type":"similarity","ratio":0.01,"country":"US"}' \ -d "origin_audience_id="
# Google Ads Customer Match — push hashed emails directly POST https://googleads.googleapis.com/v15/customers//offlineUserDataJobs:create { "job": { "type": "CUSTOMER_MATCH_USER_LIST", "customerMatchUserListMetadata": { "userList": "customers/123/userLists/Setoworks_Lookalike" } } } # Then add operations from Setoworks seed (each row = sha256 email) POST .../offlineUserDataJobs/:addOperations { "operations": [{"create":{"userIdentifiers":[{"hashedEmail":"8b2e1d4f..."}]}}, ...] }
# TikTok Ads — File Custom Audience POST https://business-api.tiktok.com/open_api/v1.3/dmp/custom_audience/file/upload/ { "advertiser_id": "123", "file_name": "setoworks_lookalike_tech_2026.csv", "file": "<multipart from /v1/lookalike/seed?format=tiktok>" } # Auto-creates Lookalike with 1% similarity in target country POST .../dmp/custom_audience/lookalike/create/ { "advertiser_id": "123", "source_audience_id": "<UPLOADED_AUDIENCE>", "region_id": ["6252001"], # US "audience_size": "NARROW" }
